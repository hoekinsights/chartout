import { ChartPreview, StorePreview } from '@/components/demo'

```python
import altair as alt
from altair.datasets import data

source = data.movies.url

chart = alt.Chart(source).mark_bar().encode(
    alt.X("IMDB Rating:Q").bin(maxbins=20).scale(domain=[1, 10]),
    alt.Y('count()'),
    alt.Color("IMDB Rating:Q").bin(maxbins=20).scale(scheme='pinkyellowgreen')
)
chart
```

<ChartPreview />

```python
import chartout

store = chartout.Store(chart)
store  # renders the widget inline in a notebook environment
```

<StorePreview />

From there you can browse products, preview your chart, adjust placement, and check out.
