There are three ways to open the store, depending on how much you want to decide for the user. Each example follows a notebook-style flow: create a chart, display it, open the store, then display the widget. The charts use the [Palmer Penguins](https://allisonhorst.github.io/palmerpenguins/) dataset.

## VizLike

If the you or the user has a chart and likes to picks the product inside the widget.

<Tabs items={['Vega-Altair', 'matplotlib']}>
  <Tab value="Vega-Altair">
    ```python
    import numpy as np
    import pandas as pd
    import altair as alt
    from altair.datasets import data

    col_class = "Species"
    cols = [
        "Beak Length (mm)",
        "Beak Depth (mm)",
        "Flipper Length (mm)",
        "Body Mass (g)",
    ]
    samples = 100

    source = data.penguins().dropna(subset=[*cols, col_class])
    # Take an even sample of each species so the chart stays light.
    frame = source.groupby(col_class, observed=True).head(15).reset_index(drop=True)

    t = np.linspace(-np.pi, np.pi, samples)
    # Normalize each dimension to (0, 1) so no single column dominates the curve.
    values = frame[cols].to_numpy(dtype=float)
    values = (values - values.min(axis=0)) / (values.max(axis=0) - values.min(axis=0))
    values = values.T

    curves = np.outer(values[0], np.ones_like(t)) / np.sqrt(2)
    for i in range(1, len(values)):
        fn = np.sin if i % 2 else np.cos
        curves += np.outer(values[i], fn(((i + 1) // 2) * t))

    rows = len(frame)
    plot_data = pd.DataFrame({
        "t": np.tile(t, rows),
        "value": curves.ravel(),
        "sample": np.repeat(np.arange(rows), samples),
        col_class: np.repeat(frame[col_class].to_numpy(), samples),
    })

    chart = alt.Chart(plot_data).mark_line().encode(
        x="t:Q",
        y=alt.Y("value:Q").title(None),
        color=f"{col_class}:N",
        detail="sample:N",
        opacity=alt.value(0.5),
    ).properties(width=468, height=182)
    chart
    ```

    <PythonChartOutput library="altair" sessionId="altair-vizlike" pattern="vizlike" />

    ```python

    store = chartout.Store(chart)
    store
    ```

## CartItem

You pre-select a specific product with your viz as item. The store opens with that item loaded in the cart.

Because you name the product, you need its id. Call `chartout.products()` to list the catalogue and pick one whose `print_ratio` suits your chart, here a square scatter for the 1:1 canvas.

<Tabs items={['Vega-Altair', 'matplotlib']}>
  <Tab value="Vega-Altair">
    We start with a square Altair chart, matching the canvas print ratio:

    ```python
    import altair as alt
    from altair.datasets import data

    penguins = data.penguins().dropna()

    chart = alt.Chart(penguins).mark_circle(opacity=0.7, size=28).encode(
        alt.X("Flipper Length (mm):Q").scale(zero=False),
        alt.Y("Body Mass (g):Q").scale(zero=False),
        alt.Color("Species:N"),
    ).properties(width=240, height=240)
    chart
    ```

## Cart

You build the entire cart with multiple items upfront. Each product can carry a different chart, sized to its print ratio.

Call `chartout.products()` first to look up each product id and its `print_ratio`: a square scatter for the 1:1 canvas, wide Andrews curves for the 18:7 mug, and a near-square density for the 6:5 mousepad.

<Tabs items={['Vega-Altair', 'matplotlib']}>
  <Tab value="Vega-Altair">
    ```python
    import numpy as np
    import pandas as pd
    import altair as alt
    from altair.datasets import data

    cols = ["Beak Length (mm)", "Beak Depth (mm)", "Flipper Length (mm)", "Body Mass (g)"]
    penguins = data.penguins().dropna(subset=[*cols, "Species"])

    # square scatter → canvas (1:1)
    canvas_chart = alt.Chart(penguins).mark_circle(opacity=0.7, size=28).encode(
        alt.X("Flipper Length (mm):Q").scale(zero=False),
        alt.Y("Body Mass (g):Q").scale(zero=False),
        alt.Color("Species:N"),
    ).properties(width=240, height=240)

    # wide Andrews curves → mug (18:7)
    frame = penguins.groupby("Species", observed=True).head(15).reset_index(drop=True)
    t = np.linspace(-np.pi, np.pi, 100)
    v = frame[cols].to_numpy(dtype=float)
    v = ((v - v.min(0)) / (v.max(0) - v.min(0))).T
    curves = np.outer(v[0], np.ones_like(t)) / np.sqrt(2)
    for i in range(1, len(v)):
        curves += np.outer(v[i], (np.sin if i % 2 else np.cos)(((i + 1) // 2) * t))
    andrews = pd.DataFrame({
        "t": np.tile(t, len(frame)),
        "value": curves.ravel(),
        "sample": np.repeat(np.arange(len(frame)), 100),
        "Species": np.repeat(frame["Species"].to_numpy(), 100),
    })
    mug_chart = alt.Chart(andrews).mark_line(opacity=0.5).encode(
        x=alt.X("t:Q").title(None),
        y=alt.Y("value:Q").title(None),
        color="Species:N",
        detail="sample:N",
    ).properties(width=468, height=182)

    # near-square density → mousepad (6:5)
    mousepad_chart = alt.Chart(penguins).mark_rect().encode(
        alt.X("Beak Length (mm):Q").bin(maxbins=20),
        alt.Y("Beak Depth (mm):Q").bin(maxbins=20),
        alt.Color("count():Q").scale(scheme="blues"),
    ).properties(width=252, height=210)

    canvas_chart | mug_chart | mousepad_chart
    ```

    import matplotlib.pyplot as plt
    from pandas.plotting import andrews_curves
    from altair.datasets import data

    cols = ["Beak Length (mm)", "Beak Depth (mm)", "Flipper Length (mm)", "Body Mass (g)"]
    penguins = data.penguins().dropna(subset=[*cols, "Species"])

    # square scatter → canvas (1:1)
    canvas_fig, ax = plt.subplots(figsize=(3, 3))
    for species, grp in penguins.groupby("Species"):
        ax.scatter(grp["Flipper Length (mm)"], grp["Body Mass (g)"], s=12, alpha=0.7, label=species)
    ax.set(xlabel="Flipper Length (mm)", ylabel="Body Mass (g)")

    # wide Andrews curves → mug (18:7)
    frame = penguins.groupby("Species", observed=True).head(15).reset_index(drop=True)

    # near-square density → mousepad (6:5)
    mousepad_fig, ax = plt.subplots(figsize=(3, 2.5))
    ax.hist2d(penguins["Beak Length (mm)"], penguins["Beak Depth (mm)"], bins=20, cmap="Blues")
